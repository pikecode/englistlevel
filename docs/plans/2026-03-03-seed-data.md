## 2. Prisma Seed 脚本

**文件**: `prisma/seed.ts`

```typescript
import { PrismaClient } from '@prisma/client'
import * as fs from 'fs'
import * as path from 'path'

const prisma = new PrismaClient()

async function main() {
  console.log('🌱 Starting database seed...')

  // 读取示例数据
  const dataPath = path.join(__dirname, '../data/sample-sentences.json')
  const rawData = fs.readFileSync(dataPath, 'utf-8')
  const { sentences } = JSON.parse(rawData)

  console.log(`📝 Found ${sentences.length} sentences to import`)

  // 批量创建句子
  for (const sentence of sentences) {
    const existing = await prisma.sentence.findFirst({
      where: {
        level: sentence.level,
        seqNo: sentence.seqNo,
      },
    })

    if (!existing) {
      await prisma.sentence.create({
        data: {
          level: sentence.level,
          seqNo: sentence.seqNo,
          enText: sentence.enText,
          zhText: sentence.zhText,
          audioUrl: sentence.audioUrl,
          ttsEnabled: true,
          status: 'active',
        },
      })
      console.log(
        `✅ Created sentence: Level ${sentence.level}, Seq ${sentence.seqNo}`
      )
    }
  }

  console.log('✨ Database seed completed!')
}

main()
  .catch((e) => {
    console.error('❌ Seed error:', e)
    process.exit(1)
  })
  .finally(async () => {
    await prisma.$disconnect()
  })
```

## 3. 导入脚本

**文件**: `scripts/import-sentences.ts`

```typescript
import { PrismaClient } from '@prisma/client'
import * as fs from 'fs'
import * as path from 'path'
import * as csv from 'csv-parse/sync'

const prisma = new PrismaClient()

interface SentenceRow {
  level: number
  seqNo: number
  enText: string
  zhText: string
  audioUrl?: string
}

/**
 * 从 CSV 文件导入句子
 * CSV 格式：level,seqNo,enText,zhText,audioUrl
 */
async function importFromCSV(filePath: string) {
  console.log(`📂 Reading CSV file: ${filePath}`)

  const fileContent = fs.readFileSync(filePath, 'utf-8')
  const records = csv.parse(fileContent, {
    columns: true,
    skip_empty_lines: true,
  }) as SentenceRow[]

  console.log(`📝 Found ${records.length} sentences to import`)

  let successCount = 0
  let skipCount = 0

  for (const record of records) {
    try {
      const existing = await prisma.sentence.findFirst({
        where: {
          level: parseInt(record.level as any),
          seqNo: parseInt(record.seqNo as any),
        },
      })

      if (existing) {
        console.log(
          `⏭️  Skipped: Level ${record.level}, Seq ${record.seqNo} (already exists)`
        )
        skipCount++
        continue
      }

      await prisma.sentence.create({
        data: {
          level: parseInt(record.level as any),
          seqNo: parseInt(record.seqNo as any),
          enText: record.enText,
          zhText: record.zhText,
          audioUrl: record.audioUrl,
          ttsEnabled: true,
          status: 'active',
        },
      })

      console.log(
        `✅ Created: Level ${record.level}, Seq ${record.seqNo} - ${record.enText}`
      )
      successCount++
    } catch (error) {
      console.error(
        `❌ Error importing Level ${record.level}, Seq ${record.seqNo}:`,
        error
      )
    }
  }

  console.log(`\n📊 Import Summary:`)
  console.log(`   ✅ Success: ${successCount}`)
  console.log(`   ⏭️  Skipped: ${skipCount}`)
  console.log(`   📝 Total: ${records.length}`)
}

/**
 * 从 JSON 文件导入句子
 */
async function importFromJSON(filePath: string) {
  console.log(`📂 Reading JSON file: ${filePath}`)

  const fileContent = fs.readFileSync(filePath, 'utf-8')
  const { sentences } = JSON.parse(fileContent)

  console.log(`📝 Found ${sentences.length} sentences to import`)

  let successCount = 0
  let skipCount = 0

  for (const sentence of sentences) {
    try {
      const existing = await prisma.sentence.findFirst({
        where: {
          level: sentence.level,
          seqNo: sentence.seqNo,
        },
      })

      if (existing) {
        console.log(
          `⏭️  Skipped: Level ${sentence.level}, Seq ${sentence.seqNo}`
        )
        skipCount++
        continue
      }

      await prisma.sentence.create({
        data: {
          level: sentence.level,
          seqNo: sentence.seqNo,
          enText: sentence.enText,
          zhText: sentence.zhText,
          audioUrl: sentence.audioUrl,
          ttsEnabled: true,
          status: 'active',
        },
      })

      console.log(
        `✅ Created: Level ${sentence.level}, Seq ${sentence.seqNo}`
      )
      successCount++
    } catch (error) {
      console.error(
        `❌ Error importing Level ${sentence.level}, Seq ${sentence.seqNo}:`,
        error
      )
    }
  }

  console.log(`\n📊 Import Summary:`)
  console.log(`   ✅ Success: ${successCount}`)
  console.log(`   ⏭️  Skipped: ${skipCount}`)
  console.log(`   📝 Total: ${sentences.length}`)
}

async function main() {
  const args = process.argv.slice(2)
  const filePath = args[0]

  if (!filePath) {
    console.error('❌ Usage: npx ts-node scripts/import-sentences.ts <file>')
    console.error('   Supported formats: .csv, .json')
    process.exit(1)
  }

  const fullPath = path.resolve(filePath)

  if (!fs.existsSync(fullPath)) {
    console.error(`❌ File not found: ${fullPath}`)
    process.exit(1)
  }

  const ext = path.extname(fullPath).toLowerCase()

  try {
    if (ext === '.csv') {
      await importFromCSV(fullPath)
    } else if (ext === '.json') {
      await importFromJSON(fullPath)
    } else {
      console.error(`❌ Unsupported file format: ${ext}`)
      process.exit(1)
    }

    console.log('\n✨ Import completed!')
  } catch (error) {
    console.error('❌ Import failed:', error)
    process.exit(1)
  } finally {
    await prisma.$disconnect()
  }
}

main()
```

## 4. package.json 配置

在 `package.json` 中添加以下脚本：

```json
{
  "scripts": {
    "seed": "prisma db seed",
    "import:csv": "ts-node scripts/import-sentences.ts",
    "import:json": "ts-node scripts/import-sentences.ts"
  },
  "prisma": {
    "seed": "ts-node prisma/seed.ts"
  }
}
```

## 5. 使用方法

### 运行种子脚本

```bash
npm run seed
```

这会导入 `data/sample-sentences.json` 中的所有句子。

### 导入 CSV 文件

```bash
npm run import:csv data/my-sentences.csv
```

CSV 文件格式：

```csv
level,seqNo,enText,zhText,audioUrl
1,1,Hello,你好,https://example.com/audio/001.mp3
1,2,Goodbye,再见,https://example.com/audio/002.mp3
```

### 导入 JSON 文件

```bash
npm run import:json data/my-sentences.json
```

JSON 文件格式：

```json
{
  "sentences": [
    {
      "level": 1,
      "seqNo": 1,
      "enText": "Hello",
      "zhText": "你好",
      "audioUrl": "https://example.com/audio/001.mp3"
    }
  ]
}
```

## 6. 数据验证

### 验证导入的数据

```bash
npx prisma studio
```

然后在浏览器中查看 `sentences` 表。

### 查询句子数量

```bash
npx prisma db execute --stdin <<EOF
SELECT level, COUNT(*) as count FROM sentences GROUP BY level ORDER BY level;
EOF
```

## 7. 生成完整数据集

对于完整的 5000 句数据，建议：

1. **准备数据文件**：将 5000 句数据整理成 CSV 或 JSON 格式
2. **分批导入**：可以分多个文件导入，避免单次导入过大
3. **验证数据**：导入后验证每个等级都有 20 句

示例：

```bash
# 导入第1-50级
npm run import:csv data/sentences-level-1-50.csv

# 导入第51-100级
npm run import:csv data/sentences-level-51-100.csv

# 验证
npx prisma studio
```
